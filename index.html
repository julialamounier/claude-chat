    <script>

        const chatContainer = document.getElementById('chat-container');

        const inputForm = document.getElementById('input-form');

        const messageInput = document.getElementById('message-input');

        

        // Substitua esta URL pela URL do seu webhook no n8n

        const N8N_WEBHOOK_URL = 'https://www.n8nwebhook.agro10x.cloud/webhook/chat';

        

        inputForm.addEventListener('submit', async (e) => {

            e.preventDefault();

            

            const message = messageInput.value.trim();

            if (!message) return;

            

            // Adiciona mensagem do usuário ao chat

            addMessage(message, 'user');

            messageInput.value = '';

            

            try {

                // Envia mensagem para o webhook n8n

                const response = await fetch(https://www.n8nwebhook.agro10x.cloud/webhook/chat, {

                    method: 'POST',

                    headers: {

                        'Content-Type': 'application/json'

                    },

                    body: JSON.stringify({ message })

                });

                

                if (!response.ok) {

                    throw new Error('Erro ao comunicar com a API');

                }

                

                const data = await response.json();

                

                // Adiciona resposta do bot ao chat

                addMessage(data.response, 'bot');

            } catch (error) {

                console.error('Erro:', error);

                addMessage('Ocorreu um erro ao processar sua mensagem.', 'bot');

            }

        });

        

        function addMessage(text, sender) {

            const messageDiv = document.createElement('div');

            messageDiv.className = `message ${sender}-message`;

            messageDiv.textContent = text;

            chatContainer.appendChild(messageDiv);

            chatContainer.scrollTop = chatContainer.scrollHeight;

        }
